\textbf{Question 1.9.25:}  
If the point $P(x,y)$ is equidistant from $A(a+b, b-a)$ and $B(a-b, a+b)$,  
prove that $bx = ay$.

\bigskip
\textbf{Solution:}

\medskip
\textbf{1. Given Data}
\begin{align}
\vec{A} &= \myvec{a+b \\ b-a}, \quad
\vec{B} = \myvec{a-b \\ a+b}, \quad
\vec{P} = \myvec{x \\ y} \label{eq:data}
\end{align}

Equivalently,
\begin{align}
\vec{A} &= \myvec{1 & 1 \\ -1 & 1}\myvec{a \\ b}, \\
\vec{B} &= \myvec{1 & -1 \\ 1 & 1}\myvec{a \\ b}.
\end{align}

\medskip
\textbf{2. Equidistant Condition}
\begin{align}
\norm{\vec{P}-\vec{A}}^2 &= \norm{\vec{P}-\vec{B}}^2 \label{eq:eqdist} \\[6pt]
(\vec{P}-\vec{A})^T(\vec{P}-\vec{A}) &= (\vec{P}-\vec{B})^T(\vec{P}-\vec{B}) \\
\vec{P}^T\vec{P} - 2\vec{A}^T\vec{P} + \vec{A}^T\vec{A} &= 
\vec{P}^T\vec{P} - 2\vec{B}^T\vec{P} + \vec{B}^T\vec{B}
\end{align}

\medskip
\textbf{3. Simplification in Matrix Form}
\begin{align}
2(\vec{B}-\vec{A})^T\vec{P} &= \vec{B}^T\vec{B} - \vec{A}^T\vec{A} \label{eq:step1}
\end{align}

Now,
\begin{align}
\vec{B}-\vec{A} &= 
\myvec{1 & -1 \\ 1 & 1}\myvec{a \\ b} -
\myvec{1 & 1 \\ -1 & 1}\myvec{a \\ b} \\[4pt]
&= \myvec{0 & -2 \\ 2 & 0}\myvec{a \\ b}. \label{eq:BA}
\end{align}

Also,
\begin{align}
\vec{A}^T\vec{A} &= \myvec{a & b}
\myvec{1 & 1 \\ -1 & 1}^T
\myvec{1 & 1 \\ -1 & 1}
\myvec{a \\ b}, \\[6pt]
\vec{B}^T\vec{B} &= \myvec{a & b}
\myvec{1 & -1 \\ 1 & 1}^T
\myvec{1 & -1 \\ 1 & 1}
\myvec{a \\ b}.
\end{align}

On simplification,
\begin{align}
\vec{B}^T\vec{B} - \vec{A}^T\vec{A} = 0. \label{eq:rhs}
\end{align}

Hence from \eqref{eq:step1},
\begin{align}
(\vec{B}-\vec{A})^T\vec{P} = 0. \label{eq:matrix-final}
\end{align}

\medskip
\textbf{4. Final Result}
Thus, the locus of $\vec{P}$ is the line
\begin{align}
\boxed{(\vec{B}-\vec{A})^T\vec{P} = 0}
\end{align}

Expanding,
\begin{align}
(\vec{B}-\vec{A})^T\vec{P} 
&= \Bigg(\myvec{0 & -2 \\ 2 & 0}\myvec{a \\ b}\Bigg)^T \myvec{x \\ y} \\[6pt]
&= \myvec{-2b & 2a}\myvec{x \\ y} \\[6pt]
&= -2bx + 2ay = 0 \\[6pt]
\Rightarrow \quad bx &= ay
\end{align}
