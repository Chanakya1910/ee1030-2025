%------------------- Frame 1 -------------------%
\begin{frame}{Question 1.9.25}
If the point $P(x,y)$ is equidistant from 
\[
A(a+b, b-a) \quad \text{and} \quad B(a-b, a+b),
\]
prove that
\[
bx = ay.
\]
\end{frame}

%------------------- Frame 2 -------------------%
\begin{frame}{Given Data}
\begin{align}
A &= \myvec{a+b \\ b-a}, \quad 
B = \myvec{a-b \\ a+b}, \quad 
P = \myvec{x \\ y} \label{eq:def}
\end{align}
\end{frame}

%------------------- Frame 3 -------------------%
\begin{frame}{Equidistant Condition}
\begin{align}
\|P-A\|^2 &= \|P-B\|^2 \label{eq:eqdist} \\[6pt]
(P-A)^T(P-A) &= (P-B)^T(P-B) \label{eq:expand1} \\[6pt]
P^TP - 2A^TP + A^TA &= P^TP - 2B^TP + B^TB \label{eq:expand2}
\end{align}
\end{frame}

%------------------- Frame 4 -------------------%
\begin{frame}{Simplification}
\begin{align}
2(B-A)^TP &= B^TB - A^TA \label{eq:step1} \\[6pt]
B-A &= \myvec{-2b \\ 2a} = 2\myvec{-b \\ a} \label{eq:diff} \\[6pt]
A^TA &= (a+b)^2 + (b-a)^2 = 2a^2 + 2b^2 \label{eq:ATA} \\[6pt]
B^TB &= (a-b)^2 + (a+b)^2 = 2a^2 + 2b^2 \label{eq:BTB} \\[6pt]
B^TB - A^TA &= 0 \label{eq:rhs0} \\[6pt]
(B-A)^TP &= 0 \label{eq:lhs}
\end{align}
\end{frame}

%------------------- Frame 5 -------------------%
\begin{frame}{Final Result}
\begin{align}
\myvec{-2b & 2a}\myvec{x \\ y} &= 0 \label{eq:prod} \\[6pt]
-2bx + 2ay &= 0 
\quad \Rightarrow \quad 
\boxed{bx = ay} \label{eq:final}
\end{align}

Thus, the locus of $P$ is the line
\[
n^Tx = 0, \quad n = \myvec{-b \\ a}.
\]
\end{frame}

#include <stdio.h>

// Function to check if bx = ay
// Returns 1 if true, 0 otherwise
int check_equation(double a, double b, double x, double y) {
    double lhs = b * x;
    double rhs = a * y;
    if ( (lhs - rhs) < 1e-6 && (rhs - lhs) < 1e-6 ) {
        return 1;  // condition satisfied
    }
    return 0;  // not satisfied
}
