\begin{frame}{Question}
Find the equation of the set of the points $P$ such that its distances 
from the points $A(3,4,-5)$ and $B(-2,1,4)$ are equal.\\
\end{frame}

\begin{frame}{Solution}
\begin{align}
\text{Let }A=(3,4,-5),\quad B=(-2,1,4),\quad P=(x,y,z).
\end{align}

Using vector notation $\|\cdot\|$ for Euclidean norm,

\begin{align}
\|P-A\|^2+\|P-B\|^2=2k^2.
\end{align}

Write $M=\dfrac{A+B}{2}$ (the midpoint of $AB$).\\

Using the identity
\begin{align}
\|P-A\|^2+\|P-B\|^2=2\|P-M\|^2+2\|A-M\|^2,
\end{align}

the condition becomes
\begin{align}
2\|P-M\|^2+2\|A-M\|^2=2k^2\\
\|P-M\|^2=k^2-\|A-M\|^2
\end{align}
\end{frame}

\begin{frame}{Section Formula(external division)}
Now compute $M$ and $\|A-M\|^2$:
\begin{align}
M=\frac{A+B}{2}=\left(\frac{3+(-2)}{2},\frac{4+1}{2},\frac{-5+4}{2}\right)
=\left(\frac{1}{2},\frac{5}{2},-\frac{1}{2}\right),
\end{align}

\begin{align}
AB=B-A=(-5,-3,9),\\
\|AB\|^2=25+9+81=115,
\end{align}

\begin{align}
\|A-M\|^2=\frac{\|AB\|^2}{4}=\frac{115}{4}.
\end{align}

Hence the locus of $P$ is the sphere with centre $M$ and radius
$\sqrt{k^2-\tfrac{115}{4}}$. Its equation is
\begin{align}
\left(x-\tfrac12\right)^2+\left(y-\tfrac52\right)^2+\left(z+\tfrac12\right)^2
= k^2-\frac{115}{4},
\end{align}
which exists (real radius) iff $k^2\ge\frac{115}{4}$
\end{frame}

import math
# Define points A and B
A = (3, 4, -5)
B = (-2, 1, 4)
# Midpoint M = (A+B)/2
M = tuple((a + b) / 2 for a, b in zip(A, B))
# Vector AB = B - A
AB = tuple(b - a for a, b in zip(B, A))
# |AB|^2
AB_sq = sum(x**2 for x in AB)
# |A-M|^2 = |AB|^2 / 4
AM_sq = AB_sq / 4
# Input k
k = float(input("Enter value of k: "))
# Radius^2
radius_sq = k*k - AM_sq
\end{lstlisting}
\end{block}
\end{frame}
