\begin{frame}{Question}
If $\vec{a}$ and $\vec{b}$ are the position vectors of $A$ and $B$ respectively, find the position vector of a point $C$ in $BA$ produced such that $BC$ = $1.5$ $BA$
\end{frame}

\begin{frame}{Solution}
$\text{Given the position vectors of } A \text{ and } B \text{ are } \vec{a} \text{ and } \vec{b}.$ \\[6pt]

Since $C$ lies on $BA$ produced and $BC = 1.5\, BA$, we have
\[
 AC = BC - BA = (1.5 - 1)\,BA = 0.5\,BA.
\]
Thus,
\[
 AC : CB = 0.5 : 1.5 = 1 : 3
\]

Therefore, $C$ divides $AB$ externally in the ratio $1:3$.
\end{frame}

\begin{frame}{Section Formula(external division)}
Writing the ratio as $k:1$ we have $k=3$. \\[8pt]

By the section formula (external division), the position vector of $C$ is
\[
\vec{OC} = \frac{k\vec{b} - \vec{a}}{k-1}, \quad (k=3).
\]

\[
\vec{OC} = \frac{3\vec{b} - \vec{a}}{2}.
\]

\[
\therefore \quad \vec{OC} = \tfrac{3}{2}\vec{b} - \tfrac{1}{2}\vec{a}.
\]
\end{frame}

\begin{block}{C Program}
\begin{lstlisting}[language=C, basicstyle=\ttfamily\small, keywordstyle=\color{blue}, frame=single]
#include <stdio.h>

int main() {
    float ax, ay, bx, by;
    printf("Enter coordinates of A (ax ay): ");
    scanf("%f %f", &ax, &ay);
    printf("Enter coordinates of B (bx by): ");
    scanf("%f %f", &bx, &by);
    int k = 3;
    float cx = (k*bx - ax) / (k - 1);
    float cy = (k*by - ay) / (k - 1);
    printf("Position vector of C is: (%.2f, %.2f)\n", cx, cy);
    return 0;
}
\end{lstlisting}
\end{block}

\begin{frame}[fragile]{Python Code}
\begin{block}{Part 2}
\begin{lstlisting}[language=Python, basicstyle=\ttfamily\small, keywordstyle=\color{blue}, frame=single]
# Section formula for external division:
# C = (k*b - a) / (k - 1)
Cx = (k*B[0] - A[0]) / (k - 1)
Cy = (k*B[1] - A[1]) / (k - 1)
C = (Cx, Cy)

print(f"Position vector of C is: {C}")
\end{lstlisting}
