\textbf{Question}:\\
If a line makes an angle of $30^\circ$ with the positive direction of the X-axis, $120^\circ$  with the positive direction of the Y-axis, then find the angle which it makes with the positive direction of the Z-axis

\textbf{Solution:} \\

Let the direction cosines of the line be
\begin{align}
\vec{d} = 
\begin{bmatrix}
\cos\alpha \\[4pt]
\cos\beta \\[4pt]
\cos\gamma
\end{bmatrix}.
\end{align}

Since it is a unit vector,
\begin{align}
\vec{d}^T \vec{d} &= \cos^2\alpha + \cos^2\beta + \cos^2\gamma = 1
\end{align}

\begin{align}
\text{Given}\quad \alpha &= 30^\circ, \quad \beta = 120^\circ
\end{align}

\begin{align}
\text{So} \quad \cos\alpha &= \cos 30^\circ = \tfrac{\sqrt{3}}{2}, \\
\cos\beta  &= \cos 120^\circ = -\tfrac{1}{2}.
\end{align}

Thus the direction vector becomes
\begin{align}
\vec{d} = 
\begin{bmatrix}
\tfrac{\sqrt{3}}{2} \\[4pt]
-\tfrac{1}{2} \\[4pt]
\cos\gamma
\end{bmatrix}.
\end{align}


Now, applying the unit vector condition:
\begin{align}
\vec{d}^T \vec{d} = 
\begin{bmatrix}
\frac{\sqrt{3}}{2} & -\frac{1}{2} & \cos\gamma
\end{bmatrix}
\begin{bmatrix}
\frac{\sqrt{3}}{2} \\[4pt]
-\frac{1}{2} \\[4pt]
\cos\gamma
\end{bmatrix}
= 1.
\end{align}

\begin{align}
\left(\tfrac{\sqrt{3}}{2}\right)^2 + \left(-\tfrac{1}{2}\right)^2 + (\cos\gamma)^2 &= 1 \\
\tfrac{3}{4} + \tfrac{1}{4} + \cos^2\gamma &= 1 \\
1 + \cos^2\gamma &= 1 \\
\cos^2\gamma &= 0
\end{align}

\begin{align}
\text{Hence,}\quad \cos\gamma = 0 \quad \Rightarrow \quad \gamma = 90^\circ
\end{align}

\begin{center}
$\boxed{\text{The line makes an angle of }90^\circ \text{ with the positive }Z\text{-axis.}}$
\end{center}
